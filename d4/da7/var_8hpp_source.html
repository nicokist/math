968  [this_vi = this->vi_, other_vi = other.vi_, prev_val]() mutable {
969  this_vi->val_ = prev_val;

971  // we have no way of detecting aliasing between this->vi_->adj_ and
972  // other.vi_->adj_, so we must copy adjoint before reseting to zero
973
